CREATE PROCEDURE insert_into_user
(user_id1 in number, account_type1 in varchar, username1 in varchar, pass1 in varchar, fcs1 in varchar, city1 in number, address1 in varchar, email1 in varchar, phone_number1 in number, activation1 in number, shop_id1 in number )
AS
    date_of_creation1 date;
    last_login_date1 date;
BEGIN
    date_of_creation1 := SYSDATE;
    last_login_date1 := SYSDATE;
	INSERT INTO USERS_ (USER_ID, ACCOUNT_TYPE, USERNAME, PASS, FCS, CITY, ADDRESS, EMAIL, PHONE_NUMBER, DATE_OF_CREATION, LAST_LOGIN_DATE, ACTIVATION)
	VALUES
	(user_id1, account_type1, username1, pass1, fcs1, city1, address1, email1, phone_number1, date_of_creation1, last_login_date1, activation1);
END;
